class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vowels = {'a','e','i','o','u'}

        total = 0
        for char in s[0:k]:
            if char in vowels: 
                total += 1

        maxVowels = total
        left = 0
            
        for right in range(k,len(s)):

            if s[right] in vowels: 
                total += 1
            if s[left] in vowels:
                total -= 1

            left += 1
            maxVowels = max(maxVowels, total)

        return maxVowels
